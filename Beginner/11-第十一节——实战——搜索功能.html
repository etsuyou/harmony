@Entry
@Component
struct LearnSearch {
  @State searchValue: string = ''

  build() {

    Column() {
      Search().onChange((val) => {
        this.searchValue = val
      })
      Button('搜索').onClick(() => {
        console.log(this.searchValue)
      })
    }
  }
}

二、需求

  1. 在页面的搜索输入框中输入搜索的文本内容，点击搜索文本按钮，最近搜索内容中展示刚输入搜索的文本内容。
  2. 点击删除图标，所有最近搜索内容清空，并展示没有搜索内容文本和相关图片。

@Entry
@Component
struct LearnSearch {
  @State searchArr: string[] = []
  @State searchVal: string = ''

  build() {
    Column() {
      Flex({ alignItems: ItemAlign.Center, justifyContent: FlexAlign.SpaceBetween }) {
        Search({ value: this.searchVal, placeholder: "搜索..." })
          .onChange((val) => {
            this.searchVal = val
          })
          .width('80%')
        Text('搜索')
          .onClick(() => {
            this.searchArr.push(this.searchVal)
            this.searchVal = ''
          }).fontSize(16).fontColor('#6094d2')
      }

      Flex({ alignItems: ItemAlign.Center, justifyContent: FlexAlign.SpaceBetween }) {
        Text('最近搜索').fontSize(20).fontWeight(FontWeight.Bolder)
        Image($r('app.media.ic_delete'))
          .onClick(() => {
            this.searchArr = []
          })
          .width(20).height(20)
      }.margin({ top: 20 })

      if (this.searchArr.length === 0) {
        Flex({justifyContent: FlexAlign.Center, alignItems: ItemAlign.Center}) {
          Image($r('app.media.ic_no_search')).width(100)
        }.layoutWeight(1)
      } else {
        Flex({ wrap: FlexWrap.Wrap }) {
          ForEach(this.searchArr, (item) => {
            Text(item)
              .fontSize(12)
              .padding('0.8%')
              .margin({ top: 10, right: 10 })
              .backgroundColor('#E5E8EA')
          })
        }
      }

    }.backgroundColor('#f1f2f5').height('100%').padding({ top: 10, left: 20, bottom: 10, right: 20 })
  }
}
