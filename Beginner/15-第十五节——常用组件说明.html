一、自定义弹窗（CustomDialog）

通过CustomDialogController类显示自定义弹窗。使用弹窗组件时，可优先考虑自定义弹窗，便于自定义弹窗的样式与内容。可用于广告、中奖、警告、软件更新等与用户交互响应操作

基本使用

/**
 * 使用CustomDialog装饰器
 * 装饰自定义弹框
 */
@CustomDialog
struct CustomDialogExample {
  // 必须设置 controller 属性 类型为 CustomDialogController
  controller: CustomDialogController
  build() {
    // 弹框内容
    Column() {
      Text('我是内容')
        .fontSize(20)
        .margin({ top: 10, bottom: 10 })
    }
  }
}


@Entry
@Component
struct LearnDialog {
  // 创建弹框控制器
  dialogController = new CustomDialogController({
    // 传入弹框组件
    builder: CustomDialogExample(),
  })

  build() {
    Row() {
      Column() {
        Button('打开弹框')
          .onClick(() => {
            this.dialogController.open()
          })
      }
      .width('100%')
    }
    .height('100%')
  }
}

弹窗的交互

/**
 * 使用CustomDialog装饰器
 * 装饰自定义弹框
 */
@CustomDialog
struct CustomDialogExample {
  // 必须设置 controller 属性 类型为 CustomDialogController
  controller: CustomDialogController

  /**
   * 取消事件
   */
  cancel: () => void

  /**
   * 确定事件
   */
  confirm: () => void

  build() {
    // 弹框内容
    Column() {
      Text('我是内容')
        .fontSize(20)
        .margin({ top: 10, bottom: 10 })
      Row() {
        Button('确定').onClick(() => {
          // 关闭弹框
          this.controller.close()
          this.confirm()
        })
        Button('取消').onClick(() => {
          // 关闭弹框
          this.controller.close()
          this.cancel()
        })
      }.justifyContent(FlexAlign.SpaceAround).width('100%')
    }.padding(20)
  }
}


@Entry
@Component
struct LearnDialog {
  // 创建弹框控制器
  dialogController = new CustomDialogController({
    // 传入弹框组件
    builder: CustomDialogExample({
      confirm: () => {
        console.log("点击确定")
      },
      cancel: () => {
        console.log("点击取消")
      }
    }),
  })

  build() {
    Row() {
      Column() {
        Button('打开弹框')
          .onClick(() => {
            this.dialogController.open()
          })
      }
      .width('100%')
    }
    .height('100%')
  }
}
