ets/pages/HomePage


import Context from '@ohos.app.ability.common';
import CommonUtils from '../common/utils/CommonUtils';
import CommonConstants from '../common/constants/CommonConstants';
import TextInputWidget from '../view/TextInputWidget';
import TextCommonWidget from '../view/TextCommonWidget';
import CustomDialogWidget from '../view/CustomDialogWidget';
import Logger from '../common/utils/Logger';
import HobbyItem from '../viewmodel/HobbyItem';

@Entry
@Component
struct HomePage {
  @State birthdate: string = '';
  @State sex: string = '';
  @State hobbies: string = '';
  private sexArray = ['男', '女'];

  customDialogController: CustomDialogController = new CustomDialogController({
    builder: CustomDialogWidget({
      hobbies: $hobbies
    }),
    alignment: DialogAlignment.Bottom,
    customStyle: true,
    offset: {
      dx: 0,
      dy: CommonConstants.DY_OFFSET
    }
  });
  
  build() {
    Column() {
      Image($r('app.media.ic_avatar'))
        .width(56)
        .height(56)
        .alignSelf(ItemAlign.Center)
        .margin({ top: 10 })
      Text('个人信息')
        .fontColor(Color.Black)
        .fontSize(16)
        .margin({ top: 10 })
        .alignSelf(ItemAlign.Center)
      TextInputWidget({
        inputImage: $r('app.media.ic_nickname'),
        hintText: '昵称'
      })
      TextCommonWidget({
        textImage: $r('app.media.ic_birthdate'),
        title: '出生日期',
        content: $birthdate,
        onItemClick: () => {
          CommonUtils.datePickerDialog((birthValue: string) => {
            this.birthdate = birthValue;
          });
        }
      })
      TextCommonWidget({
        textImage: $r('app.media.ic_sex'),
        title: '性别',
        content: $sex,
        onItemClick: () => {
          CommonUtils.textPickerDialog(this.sexArray, (sexValue: string) => {
            this.sex = sexValue;
          });
        }
      })
      TextInputWidget({
        inputImage: $r('app.media.ic_signature'),
        hintText: '个人签名'
      })
      TextCommonWidget({
        textImage: $r('app.media.ic_hobbies'),
        title: '兴趣爱好（多选）',
        content: $hobbies,
        onItemClick: () => {
          this.customDialogController.open();
        }
      })
    }
    .backgroundColor($r('app.color.column_background_color'))
    .width(CommonConstants.FULL_WIDTH)
    .height(CommonConstants.FULL_HEIGHT)
  }
}

ets/view/CustomDialogWidget



import HobbyItem from '../viewmodel/HobbyItem';

@CustomDialog
export default struct CustomDialogWidget {
  @State hobbyItems: HobbyItem[] = [{ label: '足球' }, { label: "乒乓球" }, { label: "篮球" }, { label: "睡觉" }];
  @Link hobbies: string;
  private controller?: CustomDialogController;

  aboutToAppear () {
    this.hobbyItems.forEach(item => {
      item.isChecked = this.hobbies.includes(item.label)
    })
  }

  build() {
    Column() {
      Text("兴趣爱好")
        .fontSize(20)
        .lineHeight(28)
        .fontWeight(500)
        .alignSelf(ItemAlign.Start)
        .margin({ left: 24 })
      List() {
        ForEach(this.hobbyItems, (itemHobby: HobbyItem) => {
          ListItem() {
            Row() {
              Text(itemHobby.label)
                .fontSize(16)
                .layoutWeight(1)
                .textAlign(TextAlign.Start)
                .fontWeight(500)
                .margin({ left: 24 })
              Toggle({ type: ToggleType.Checkbox, isOn: itemHobby.isChecked })
                .onChange((isCheck) => {
                  itemHobby.isChecked = isCheck;
                })
                .width(20)
                .height(20)
                .margin({ right: 24 })
            }
            .height(22)
            .margin({
              top: 13,
              bottom: 13
            })
          }
        }, (itemHobby: HobbyItem) => JSON.stringify(itemHobby.label))
      }
      .margin({
        top: 14,
        bottom: 8
      })
      .divider({
        strokeWidth: 0.5,
        color: '#0D182431'
      })
      .width('100%')
      .height(240)

      Row() {
        Button('取消')
          .dialogButtonStyle()
          .onClick(() => {
            this.controller?.close();
          })
        Blank()
          .backgroundColor('#F2F2F2')
          .width(1)
          .height(25)
        Button('确定')
          .dialogButtonStyle()
          .onClick(() => {
            this.controller?.close();
            this.hobbies = this.hobbyItems.filter(item => item.isChecked).map(item => item.label).join(',')
          })
      }
    }
    .width('94%')
    .padding({
      top: 14,
      bottom: 16
    })
    .borderRadius(32)
    .backgroundColor(Color.White)
  }
}

@Extend(Button) function dialogButtonStyle() {
  .fontSize(16)
  .fontColor('#007DFF')
  .layoutWeight(1)
  .backgroundColor(Color.White)
  .width('100%')
  .height(40)
}

ets/view/TextCommonWidget


import CommonConstants from '../common/constants/CommonConstants';

@Component
export default struct TextCommonWidget {
  @Link content: string;
  private textImage?: Resource;
  private title?: string;
  onItemClick = () => {
  };

  build() {
    Row() {
      Image(this.textImage !== undefined ? this.textImage : '')
        .width(24)
        .height(24)
        .margin({ left: 12 })
      Text(this.title)
        .fontSize(16)
        .margin({ left: 12 })
        .height(CommonConstants.FULL_HEIGHT)
      Text(this.content)
        .fontSize(16)
        .textAlign(TextAlign.End)
        .margin({
          left: 16,
          right: 7
        })
        .layoutWeight(1)
      Image($r('app.media.ic_arrow'))
        .width(12)
        .height(24)
        .margin({ right: 14 })
    }
    .margin({ top: 24 })
    .borderRadius(24)
    .backgroundColor(Color.White)
    .width('94%')
    .height(64)
    .onClick(this.onItemClick)
  }
}

@Component
export default struct TextInputWidget {
  private inputImage?: Resource;
  private hintText?: string;

  build() {
    Row() {
      Image(this.inputImage !== undefined ? this.inputImage : '')
        .width(24)
        .height(24)
        .margin({ left: 12 })
      TextInput({ placeholder: this.hintText })
        .fontSize(16)
        .padding({ left: 12 })
        .backgroundColor(Color.White)
        .fontWeight(FontWeight.Normal)
        .fontStyle(FontStyle.Normal)
        .fontColor(Color.Black)
        .margin({ right: 32 })
        .layoutWeight(1)
        .height(48)
    }
    .margin({ top: 24 })
    .borderRadius(24)
    .backgroundColor(Color.White)
    .width('94%')
    .height(64)
  }
}
