一、概念

@State装饰的变量，或称为状态变量，一旦变量拥有了状态属性，就和自定义组件的渲染绑定起来。当状态改变时，UI会发生对应的渲染改变 @State装饰的变量，与声明式范式中的其他被装饰变量一样，是私有的，只能从组件内部访问，在声明时必须指定其类型和本地初始化。初始化也可选择使用命名参数机制从父组件完成初始化。

二、State初始化

默认初始化

@Entry
@Component
struct LearnState {
  // 张三就是 name的初始值
  @State name: string = '张三'

  build() {
    Column() {
      Text(this.name)

    }
  }
}

传入的参数作为初始值



@Component
struct LearnState {
  /**
   * 当前name的值为父组件传过来的值 李四
   
   */
  @State name: string = '张三'

  build() {
    Column() {
      Text(this.name)
      Button('更新子组件的name')
        .onClick(() => {
          /**
           * * 注意：name改变后 父组件的值不会发生变化
           * 也就是 该值并不具备双向绑定
           */
          this.name = '王五'
        })
    }
  }
}

@Entry
@Component
struct LearnStateParent {
  // 如果传递的值在子组件已经被@State装饰 那么不能再使用@State装饰的值覆盖
  // @State parentName: string = '李四'

  parentName = '李四'
  build() {
    Column() {
      LearnState({ name: this.parentName })
      Button('查看父组件的name')
        .onClick(() => {
          // 父组件的值还是李四
          console.log(this.parentName)
        })
    }
  }
}

三、基本使用

装饰基本数据类型

实现点击按钮+1功能

@Entry
@Component
struct LearnState {
  /**
   * 使用@State装饰器 修饰num
   * 让num 属性具备响应式
   * 也就是就该后视图会发生变化
   *
   * 注意：@State装饰的属性必须定义类型
   */
  @State num: number = 1


  add () {
    /***
     * 直接对数据进行修改
     * 视图就会相应发生发辫
     */
    this.num += 1
  }

  build() {
    Column() {
      Button('+1')
        .onClick(() => this.add())
      Text(this.num + '')
    }
  }
}

装饰引用数据类型

@State只会对引用数据类型的第一层赋予响应式数据的能力，嵌套的属性不具备响应式


interface Info {
  name: string
  child?: Info
}


@Entry
@Component
struct LearnState {
  @State info: Info = {
    name: "张三",
    child: {
      name: "张三的儿子"
    }
  }


  updateName () {
    // 有效
    this.info.name = '李四'
    // 无效
    this.info.child.name = '李四的儿子'
  }

  build() {
    Column() {
      Button('修改名字')
        .onClick(() => this.updateName())
      Text(this.info.child.name)
      Text(this.info.name)
    }
  }
}
