      select: false

  updateNum(index: number, type: number) {

    const goodsItem = this.goods[index]

    if (goodsItem.num + type === 0) {
      return
    }

    goodsItem.num += type
    this.goods = [...this.goods]
  }

  allSelectChange() {
    this.isAllSelect = !this.isAllSelect
    this.goods.forEach(item => {
      item.select = this.isAllSelect
    })
  console.log("全选", JSON.stringify(this.isAllSelect))
    this.goods = [...this.goods]
  }

  singleSelect (index: number) {
    const goodsItem = this.goods[index]
    goodsItem.select = !goodsItem.select

    this.isAllSelect = this.goods.every(item => item.select)
    console.log(JSON.stringify(this.goods), JSON.stringify(this.isAllSelect))
    this.goods = [...this.goods]
  }

  build() {
    Column() {
      Row() {
        Checkbox().onClick(() => this.allSelectChange()).select(this.isAllSelect)
        Text('全选')
      }

      ForEach(this.goods, (item, index) => {
        Column() {
          Row() {
            Checkbox().select(item.select).onClick(() => {
              this.singleSelect(index)
            })
            Text('商品名：' + item.name)
            Text('价格：' + item.price)
          }

          Row() {
            Button('+1').onClick(() => {
              this.updateNum(index, 1)
            })
            Text(item.num + '')
            Button('-1').onClick(() => {
              this.updateNum(index, -1)
            })
            Text('小计：' + item.num * item.price)
          }
        }.border({ width: 1, color: Color.Red }).margin(20).padding(20)
      })

      Text('总价：' + this.goods.filter(item => item.select).reduce((total, item) => total += item.num * item.price, 0))
    }
  }
}

面向对象的写法

viewmodel/GoodsItem


@Observed
export default class GoodsItem {
  /**
   * 商品名
   */
  name: string

  /**
   * 商品价格
   */
  price: number

  /**
   * 当前选择状态
   */
  select: boolean

  /**
   * 数量
   */
  num: number

  constructor(name: string, price: number, select = false, num = 1) {
    this.name = name

    this.price = price

    this.select = select

    this.num = num
  }

  /**
   * 更新数量
   * @param num 1 还是 -1
   */
  updateNum (num: number) {
    // 数量不能为0
    if (this.num + num === 0) {
      return
    }
    this.num += num
  }

  /**
   * 单选操作
   * @param select 当前选中状态
   */
  singleSelect () {
    this.select = !this.select
  }
}

viewmodel/GoodsModel

import GoodsItem from './GoodsItem'


@Observed
export default class GoodsModel {
  goods: GoodsItem[] = [
    new GoodsItem('羊肉串', 10),
    new GoodsItem('猪肉串', 5),
    new GoodsItem('鸡翅', 15)
  ]
  isAllSelect: boolean = false
  total: number = 0

  handleAllSelect() {
    this.isAllSelect = !this.isAllSelect
    this.goods.forEach(item => {
      item.select = this.isAllSelect
    })
    this.sum()
  }

  sum() {
    this.total = this.goods.filter(item => item.select).reduce((total, item) => total += item.num * item.price, 0)
  }

  singleSelect () {
    this.updateAllSelectStatus()
    this.sum()
  }

  updateAllSelectStatus () {
    this.isAllSelect = this.goods.every(item => item.select)
  }
}

viewmodel/GoodsWidget

import GoodsItem from '../viewmodel/GoodsItem'

@Component
export default struct GoodsWidget {
  @ObjectLink item: GoodsItem

  onSingleSelect = () => {}

  build() {
    Column() {
      Row() {
        Checkbox().select(this.item.select).onClick(() => {
          this.item.singleSelect()
          this.onSingleSelect()
        })
        Text('商品名：' + this.item.name)

        })

}

pages/ShopCart

import GoodsWidget from '../view/GoodsWidget'
import GoodsItem from '../viewmodel/GoodsItem'
import GoodsModel from '../viewmodel/GoodsModel'

@Entry
@Component
struct ShopCart {
  @State goodsModel: GoodsModel = new GoodsModel()

  build() {
    Column() {
      Row() {
        Checkbox().onClick((select) => this.goodsModel.handleAllSelect()).select(this.goodsModel.isAllSelect)
        Text('全选')
      }

      ForEach(this.goodsModel.goods, (item: GoodsItem) => {
        GoodsWidget({ item, onSingleSelect: () => {
            this.goodsModel.singleSelect()
        }})
      })

      Text('总价：' + this.goodsModel.total)
    }
  }
}
